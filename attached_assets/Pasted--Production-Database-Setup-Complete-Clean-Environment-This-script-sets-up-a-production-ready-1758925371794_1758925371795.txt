-- Update hire_bookings table structure if needed
DO $$
BEGIN
    -- Add columns if they don't exist
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'hire_bookings' AND column_name = 'service_type') THEN
        ALTER TABLE hire_bookings ADD COLUMN service_type TEXT DEFAULT 'car' CHECK (service_type IN ('car', 'boat'));
    END IF;
    
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'hire_bookings' AND column_name = 'service_name') THEN
        ALTER TABLE hire_bookings ADD COLUMN service_name TEXT;
    END IF;
    
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'hire_bookings' AND column_name = 'images') THEN
        ALTER TABLE hire_bookings ADD COLUMN images TEXT[] DEFAULT '{}';
    END IF;
    
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'hire_bookings' AND column_name = 'features') THEN
        ALTER TABLE hire_bookings ADD COLUMN features TEXT[] DEFAULT '{}';
    END IF;
END $$;

-- Create hire_services table for managing hire inventory
CREATE TABLE IF NOT EXISTS hire_services (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  type TEXT NOT NULL CHECK (type IN ('car', 'boat')),
  description TEXT,
  price DECIMAL(10,2) NOT NULL,
  duration TEXT,
  capacity TEXT,
  images TEXT[] DEFAULT '{}',
  features TEXT[] DEFAULT '{}',
  is_available BOOLEAN DEFAULT TRUE,
  is_featured BOOLEAN DEFAULT FALSE,
  rating DECIMAL(2,1) DEFAULT 0.0,
  review_count INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Enable RLS on hire_services
ALTER TABLE hire_services ENABLE ROW LEVEL SECURITY;

-- Hire services policies

-- Create indexes for better performance
CREATE INDEX IF NOT EXISTS idx_products_category_active ON products(category_id, is_active);
CREATE INDEX IF NOT EXISTS idx_products_featured_active ON products(is_featured, is_active);
CREATE INDEX IF NOT EXISTS idx_products_price ON products(price);
CREATE INDEX IF NOT EXISTS idx_categories_active_sort ON categories(is_active, sort_order);
CREATE INDEX IF NOT EXISTS idx_properties_available_featured ON real_estate_properties(is_available, is_featured);
